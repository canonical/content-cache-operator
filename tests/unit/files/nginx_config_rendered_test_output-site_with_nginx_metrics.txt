proxy_cache_path /var/lib/nginx/proxy/site_with_nginx_metrics use_temp_path=off levels=1:2 keys_zone=site_with_nginx_metrics-cache:10m max_size=1g;

server {
    server_name site_with_nginx_metrics;
    listen 127.0.0.1:6080;

    port_in_redirect off;
    absolute_redirect off;

    location / {
        proxy_pass http://localhost:8080;
        proxy_set_header Host "site_with_nginx_metrics";
        proxy_cache site_with_nginx_metrics-cache;
        proxy_cache_background_update on;
        proxy_cache_lock on;
        proxy_cache_min_uses 1;
        proxy_cache_revalidate on;
        proxy_cache_use_stale error timeout updating http_500 http_502 http_503 http_504;
        proxy_cache_valid 200 1d;

        log_by_lua_block {
          http_request_total:inc(1, {ngx.var.server_name, ngx.var.status})
          if ngx.var.upstream_cache_status == "HIT" then
              cache_request_hit_total:inc(1, {ngx.var.server_name})
          end
        }
    }


    access_log /var/log/nginx/site_with_nginx_metrics-access.log content_cache;
    error_log /var/log/nginx/site_with_nginx_metrics-error.log;
}
