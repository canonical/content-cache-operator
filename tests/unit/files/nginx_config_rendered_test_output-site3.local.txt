proxy_cache_path /var/lib/nginx/proxy/site3.local use_temp_path=off levels=1:2 keys_zone=site3-cache:10m max_size=1g;

server {
    server_name site3.local;
    listen 6083;

    location / {
        proxy_pass http://localhost:8083;
        proxy_set_header Host "site3.local";
        proxy_cache site3;
        proxy_cache_valid 200 1d;
        proxy_cache_min_uses 5;
        proxy_cache_revalidate on;
        proxy_cache_use_stale error timeout updating http_500 http_502 http_503 http_504;
        proxy_cache_background_update on;
        proxy_cache_lock on;
    }

    access_log /var/log/nginx/site3.local-access.log;
    error_log /var/log/nginx/site3.local-error.log;
}
