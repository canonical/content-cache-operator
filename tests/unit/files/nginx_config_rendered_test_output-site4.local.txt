proxy_cache_path /var/lib/nginx/proxy/site4.local use_temp_path=off levels=1:2 keys_zone=site4-cache:10m max_size=1g;

server {
    server_name site4.local;
    listen 127.0.0.1:6083;

    location / {
        proxy_pass http://localhost:8083;
        proxy_set_header Host "site4.local";
        proxy_cache site4-cache;
        proxy_cache_background_update on;
        proxy_cache_lock on;
        proxy_cache_min_uses 1;
        proxy_cache_revalidate on;
        proxy_cache_use_stale error timeout updating http_500 http_502 http_503 http_504;
        proxy_cache_valid 200 1d;
    }

    location / {
        autoindex on;
    }

    location /ubuntu/pool/ {
        autoindex on;
        auth_request /auth;
    }


    access_log /var/log/nginx/site4.local-access.log content_cache;
    error_log /var/log/nginx/site4.local-error.log;
}
